int main(void)
{
    puts("Welcome to your MPG Calculator.");

    float totmiles;
    float totgals;

    while (1)
    {
        float gals;
        printf("Enter gallons (-1 to exit): ");
        scanf(gals);

        if gals == 1
        {
            break;
        }

        float miles;
        printf("Enter miles: ");
        scanf(miles);

        printf("MPG for trip: " (miles / gals));
        increment totmiles and totgals
    }
    printf("Total miles, total gallons, total MPG");
}
